52  bool PMT::writeSection(Section& section, ui8 cur_sec, ui16 &sec_bytes) const
53  {
54  ui8 *prog_info_len_pos = 0;
55  ui16 d_len, prog_info_len = 0;
56  bool done = false, exit = false;
57 
58  while (!exit)
59  {
60  switch (run.op_state)
61  {
62  case INIT:
63  // associate the iterators to the list
64  run.pd_iter = prog_desc.begin();
65  run.es_iter = es_list.begin();
66  run.op_state = WRITE_HEAD;
67 
68  case WRITE_HEAD:
69  // common data for every section
70  // if table's length is > max_section_len, we'll
71  // overwrite the section length later on
72  writeSectionHeader(section);
73 
74  // section count information
75  section.set08Bits(cur_sec);
76 
77  // we don't know the last section number yet, so
78  // for now set it to 0,
79  section.set08Bits(0);//last_sec_num);
80 
81  section.set16Bits( rbits(0xe000) | pcr_pid );
82 
83  // save the position for the network_desc_loop_len
84  prog_info_len_pos = section.getCurDataPosition();
85 
86  // for now, set it to something.. this is necessary to
87  // increment the internal data pointer of the buffer
88  section.set16Bits( 0 );
89 
90  sec_bytes = BASE_LENGTH; // the minimum section size
91  run.op_state = (!run.pd ? GET_PROG_DESC : WRITE_PROG_DESC);
92  break;
93 
94  case GET_PROG_DESC:
95  // update the prog_info_len.. we always do this
96  // and if we don't add any, it is set to 0 anyways
97  section.set16Bits(prog_info_len_pos,
98  rbits(~LEN_MASK) |
99  (prog_info_len & LEN_MASK) );
100 
101  if (!run.d_done) {
102  // fetch the next program descriptor
103  if (run.pd_iter != prog_desc.end()) {
104  run.pd = (*run.pd_iter++).get();
105 
106  // check if we can fit it in this section
107  if (sec_bytes + run.pd->length() > getMaxDataLen()) {
108  // we can't.. return so we can get a new section
109  // we'll add it when we come back
110  run.op_state = WRITE_HEAD;
111  exit = true;
112  }
113  else // we found one to write
114  run.op_state = WRITE_PROG_DESC;
115  break;
116  }
117  else {
118  run.pd = nullptr;
119  run.d_done = true;
120  // fall through
121  }
122  }
123 
124  // done with all program descriptors.. move on to the
125  // transport streams
126  run.op_state = (!run.es ? GET_XPORT_STREAM : WRITE_XPORT_STREAM);
127  break;
128 
129  case WRITE_PROG_DESC:
130  // add the network descriptor
131  run.pd->buildSections(section);
132 
133  d_len = run.pd->length();
134  sec_bytes += d_len;
135  prog_info_len += d_len;
136 
137  // try to add another one
138  run.op_state = GET_PROG_DESC;
139  break;
140 
141  case GET_XPORT_STREAM:
142  // fetch a transport stream
143  if (run.es_iter != es_list.end()) {
144  run.es = (*run.es_iter++);
145 
146  // first, check if it has any descriptors.. we'll try to fit
147  // at least one
148  if (!run.es->descriptors.empty()) {
149  const Descriptor *d = run.es->descriptors.front().get();
150 
151  // check the size with the descriptor
152  if ( (sec_bytes + PMT::ElementaryStream::BASE_LEN + d->length()) >
153  getMaxDataLen() ) {
154  // won't fit.. wait until the next section
155  run.op_state = WRITE_HEAD;
156  exit = true;
157  break;
158  }
159  }
160  else {
161  // check if this XS with no descs will fit here
162  if ( (sec_bytes + PMT::ElementaryStream::BASE_LEN) > getMaxDataLen() ) {
163  // nope.. wait also
164  run.op_state = WRITE_HEAD;
165  exit = true;
166  break;
167  }
168  }
169  // all is ok.. add it
170  run.op_state = WRITE_XPORT_STREAM;
171  }
172  else {
173  // no more network descriptors or transport streams, so
174  // all sections are done!
175  run = Context();
176  exit = done = true;
177  break;
178  }
179  break;
180 
181  case WRITE_XPORT_STREAM:
182  // finally write it
183  if (!(*run.es).write_section(section, getMaxDataLen(), sec_bytes)) {
184  run.op_state = WRITE_HEAD;
185  exit = true;
186  break;
187  }
188  run.op_state = GET_XPORT_STREAM;
189  break;
190  }
191  }
192  return done;
193  }
