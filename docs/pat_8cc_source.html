46  // writes to the stream
47  bool PAT::writeSection(Section& section, ui8 cur_sec, ui16 &sec_bytes) const
48  {
49  bool done = false;
50  bool exit = false;
51 
52  while (!exit)
53  {
54  switch (run.op_state)
55  {
56  case INIT:
57  run.p_iter = program_list.begin();
58  run.op_state = WRITE_HEAD;
59  // fall through
60 
61  case WRITE_HEAD:
62  // common data for every section
63  // if table's length is > available space, we'll
64  // overwrite the section length later on
65  writeSectionHeader(section);
66 
67  // section count information
68  section.set08Bits(cur_sec);
69 
70  // we don't know the last section number yet, so
71  // for now set it to 0,
72  section.set08Bits(0);
73 
74  sec_bytes = BASE_LENGTH; // the minimum section size
75  run.op_state = (!run.p ? GET_PROGRAM : WRITE_PROGRAM);
76  break;
77 
78  case GET_PROGRAM:
79  // fetch the next service
80  if (run.p_iter != program_list.end()) {
81  run.p = &(*run.p_iter++);
82 
83  // does adding it exceed capacity?
84  if ( (sec_bytes + Program::BASE_LEN) > getMaxDataLen() ) {
85  // yes
86  run.op_state = WRITE_HEAD;
87  exit = true;
88  break;
89  }
90  // we can add it
91  run.op_state = WRITE_PROGRAM;
92  }
93  else {
94  // done with all programs.. all sections are done!
95  run = Context();
96  exit = done = true;
97  break;
98  }
99  break;
100 
101  case WRITE_PROGRAM:
102  section.set16Bits(run.p->number);
103  section.set16Bits( rbits(0xe000) | run.p->pid );
104 
105  sec_bytes += Program::BASE_LEN;
106  run.op_state = GET_PROGRAM;
107  break;
108  }
109  }
110  return done;
111  }
