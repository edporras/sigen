154  void SSUDataBroadcastIdDesc::buildSections(Section& s) const
155  {
156  _DataBroadcastIdDesc::buildSections(s);
157 
158  s.set08Bits( OUI_data_len );
159 
160  for (const auto &oui : oui_list) {
161  s.set24Bits( oui.OUI );
162  s.set08Bits( rbits(0xf0) | oui.update_type );
163  s.set08Bits( rbits(0xc0) | (oui.update_versioning_flag << 5) | oui.update_version );
164  s.set08Bits( oui.selector_bytes.size() );
165  if ( oui.selector_bytes.size() )
166  s.setBits( oui.selector_bytes );
167  }
168 
169  s.setBits( private_data );
170  }
