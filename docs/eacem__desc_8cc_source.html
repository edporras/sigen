41  {
42  // check if we can add another to the loop

44  return false;

52  // write the binary data
53  //
54  void LogicalChannelDesc::buildSections(Section &s) const
55  {
56  Descriptor::buildSections(s);
57 
58  for (const auto &channel : channel_list) {
59  s.set16Bits( channel.sid );
60  s.set16Bits( (channel.visible_service_flag << 15) |
61  (rbits(0x1f) << 10) |
62  (channel.logical_channel_number & 0x03ff) );
63  }
64  }
