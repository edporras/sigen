74  //
75  // handles writing the data to the stream. return true if the table
76  // is done (all sections are completed)
77  //
78  bool NIT_BAT::writeSection(Section& section, ui8 cur_sec, ui16& sec_bytes) const
79  {
80  ui8 *nd_loop_len_pos = 0, *ts_loop_len_pos = 0;
81  ui16 d_len, net_desc_len = 0, ts_loop_len = 0;
82  bool done = false, exit = false;
83 
84  while (!exit)
85  {
86  switch (run.op_state)
87  {
88  case INIT:
89  // associate the iterators to the lists.. once they reach the
90  // end, they'll take care to reset themselves
91  run.nd_iter = descriptors.begin();
92  run.ts_iter = xs_list.begin();
93  run.op_state = WRITE_HEAD;
94 
95  case WRITE_HEAD:
96  // common data for every section
97  // if table's length is > max_section_len, we'll
98  // overwrite the section length later on
99  writeSectionHeader(section);
100 
101  // section count information
102  section.set08Bits(cur_sec);
103 
104  // we don't know the last section number yet, so
105  // for now set it to 0,
106  section.set08Bits(0);//last_sec_num);
107 
108  // save the position for the descriptors_loop_len
109  nd_loop_len_pos = section.getCurDataPosition();
110 
111  // for now, save the original value.. this is necessary to
112  // increment the internal data pointer of the buffer
113  section.set16Bits(descriptors.loop_length());
114 
115  sec_bytes = BASE_LENGTH; // the minimum section size
116  run.op_state = (!run.nd ? GET_NET_DESC : WRITE_NET_DESC);
117  break;
118 
119  case GET_NET_DESC:
120  // update the descriptors_len.. we always do this
121  // and if we don't add any, it is set to 0 anyways
122  section.set16Bits(nd_loop_len_pos,
123  rbits(~LEN_MASK) |
124  (net_desc_len & LEN_MASK) );
125 
126  if (!run.nd_done) {
127  // fetch the next network descriptor
128  if (run.nd_iter != descriptors.end()) {
129  run.nd = (*run.nd_iter++).get();
130 
131  // check if we need to move to a new section -
132  // either a forced break or we can't fit the
133  // descriptor in this section
134  if (run.nd->type() == UtilityDesc::SECTION_END_MARK ||
135  (sec_bytes + run.nd->length() > getMaxDataLen())) {
136  // if the section is filled with net
137  // descriptors, we must sitll write an empty XS
138  // loop length!
139  ts_loop_len_pos = section.getCurDataPosition();
140  section.set16Bits( 0 );
141 
142  // we can't.. return so we can get a new section
143  // we'll add it when we come back
144  run.op_state = WRITE_HEAD;
145 
146  // if using a utility desc to force a break,
147  // clear the pointer so the next one is fetched
148  if (run.nd->type() == UtilityDesc::SECTION_END_MARK) {
149  run.nd = nullptr;
150  }
151 
152  exit = true;
153  }
154  else // found one
155  run.op_state = WRITE_NET_DESC;
156  break;
157  }
158  else {
159  run.nd_done = true;
160  run.nd = nullptr;
161  // fall through
162  }
163  }
164 
165  // done with all network descriptors.. move on to the
166  // transport streams
167  run.op_state = WRITE_XPORT_LOOP_LEN;
168  break;
169 
170  case WRITE_NET_DESC:
171  // add the network descriptor
172  run.nd->buildSections(section);
173 
174  d_len = run.nd->length();
175  sec_bytes += d_len;
176  net_desc_len += d_len;
177 
178  // try to add another one
179  run.op_state = GET_NET_DESC;
180  break;
181 
182  case WRITE_XPORT_LOOP_LEN:
183  // save the location for the transport stream loop length
184  ts_loop_len_pos = section.getCurDataPosition();
185  // again, this will be overwritten, but is necessary to
186  // increment the internal buffer pointer
187 
188  section.set16Bits( 0 ); //xport_stream_loop_length);
189 
190  // if we were looking at one already, don't get a new xport stream
191  run.op_state = (!run.ts ? GET_XPORT_STREAM : WRITE_XPORT_STREAM);
192  break;
193 
194  case GET_XPORT_STREAM:
195 
196  // fetch a transport stream
197  if (run.ts_iter != xs_list.end()) {
198  run.ts = *(run.ts_iter++);
199 
200  // first, check if it has any descriptors.. we'll try to fit
201  // at least one
202  if (!run.ts->descriptors.empty()) {
203  // check the size with the descriptor
204  if ( (sec_bytes + XportStream::BASE_LEN + run.ts->descriptors.front()->length()) >
205  getMaxDataLen() ) {
206  // won't fit.. wait until the next section
207  run.op_state = WRITE_HEAD;
208  exit = true;
209  break;
210  }
211  }
212  else {
213  // check if this XS with no descs will fit here
214  if ( (sec_bytes + XportStream::BASE_LEN) > getMaxDataLen() ) {
215  // nope.. wait also
216  run.op_state = WRITE_HEAD;
217  exit = true;
218  break;
219  }
220  }
221  // all is ok.. add it
222  run.op_state = WRITE_XPORT_STREAM;
223  }
224  else {
225  // no more network descriptors or transport streams, so
226  // all sections are done!
227  run = Context();
228  exit = done = true;
229  break;
230  }
231  break;
232 
233  case WRITE_XPORT_STREAM:
234  // finally write it
235  if (!(*run.ts).write_section(section, getMaxDataLen(), sec_bytes, &ts_loop_len)) {
236  run.op_state = WRITE_HEAD;
237  exit = true;
238  break;
239  }
240  run.op_state = GET_XPORT_STREAM;
241  break;
242  }
243  }
244 
245  // save the xport stream length we've calculated so far
246  section.set16Bits( ts_loop_len_pos,
247  rbits(~LEN_MASK) |
248  (ts_loop_len & LEN_MASK) );
249  return done;
250  }

275  NETWORK_ID_S);
276  } else {
277  // BAT

279  }
280 
281  // reserved bits
282  identStr(o, RESERVED_FU_S, 0xf);
283 
284  // descriptors
285  identStr(o,
286  ((getId() == NIT_BAT::BAT_TID) ? BOUQUET_DESC_LEN_S : NETWORK_DESC_LEN_S),
287  descriptors.loop_length());
288  o << std::endl;
289 
290  descriptors.dump(o);
291 
292  // transport streams

299  // dumps the transport stream data to stdout
300  //
301  void NIT_BAT::dumpXportStreams(std::ostream& o) const
302  {
303  // loop length
304  o << std::hex;
305  identStr(o, RESERVED_FU_S, 0xf);
306 
307  ui16 xsl_len = std::accumulate(xs_list.begin(), xs_list.end(),
308  XportStream::BASE_LEN * xs_list.size(),
309  [](int len, const auto xs) { return len + xs->descriptors.loop_length(); });
310  identStr(o, XS_LOOP_LEN_S, xsl_len);
311  o << std::endl;
312 
313  // display each transport stream's data & descriptors
314  incOutLevel();
315  for (const ListItem* item : xs_list) {
316  const XportStream* xs = dynamic_cast<const XportStream*>(item);
317  headerStr(o, XPORT_STREAM_S);
318 
319  identStr(o, XPORT_STREAM_ID_S, xs->id);
320  identStr(o, ORIG_NETWORK_ID_S, xs->original_network_id);
321  identStr(o, RESERVED_FU_S, 0xf);
322  identStr(o, DESC_LEN_S, xs->descriptors.loop_length());
323  o << std::endl;
324 
325  // dump the descriptors (inherited method)
326  xs->descriptors.dump(o);
327  }
328  decOutLevel();
329  }
