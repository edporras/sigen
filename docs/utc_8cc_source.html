87  // ----------------------------------------------------------------------
88  // global conversion routines mjd <-> M D Y
89  //
90  ui16 UTC_priv::date2MJD(ui16 M, ui16 D, ui16 Y)
91  {
92  ui16 mjd, L = 0;
93 
94  // remove the leading '19'
95  if (Y > 1900)
96  Y -= 1900;
97 
98  // from the formula in pg 71 ETS 300 468 jan 96
99  if (M == 1 || M == 2)
100  L = 1;
101 
102  mjd = 14956 + (ui16) D +
103  (ui16) (((float) Y - (float) L) * (float) 365.25) +
104  (ui16) (((float) M + (float) 1 + (float) L * (float) 12) *
105  (float) 30.6001);
106  return mjd;
107  }

109  void UTC_priv::MJD2Date(ui16 mjd, ui16 &M, ui16 &D, ui16 &Y)
110  {
111  ui16 mo, yr, K = 0;
112  double Mp;
113 
114  const double DPY = 365.25;
115 
116  // parse the MJD.. do the math
117  yr = (ui16) ((mjd - 15078.2) / DPY);
118  Mp = mjd - 14956.1 - (ui16)(yr * DPY);
119  mo = (ui16) (Mp / 30.6001);
120  D = mjd - 14956 - (ui16)(yr * DPY) - (ui16)(mo * 30.6001);
121 
122  if (mo == 14 || mo == 15)
123  K = 1;
124 
125  Y = yr + K + 1900;
126  M = mo - 1 - K * 12;
127  }
