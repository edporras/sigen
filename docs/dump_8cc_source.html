363  //
364  // dumps a single data line
365  //
366  static void dumpDataLine(std::ostream& o, const ui8* data, ui8 length)
367  {
368  uint_fast8_t i;
369 
370  o << std::hex;
371 
372  // display the data in byte mode
373  for (i = 0; i < length; i++)
374  o << std::setw(2) << static_cast<ui16>(data[i]) << " ";
375 
376  // fill any empty spaces (if length < LINE_WIDTH)
377  for (; i < LINE_WIDTH; i++)
378  o << " ";
379 
380  // display the data in char mode
381  o << " ";
382 
383  for (i = 0; i < length; i++) {
384  char c = data[i];
385 
386  // map unprintables to '.'
387  if ((c < ' ') || (c > '~'))
388  c = '.';
389 
390  o << std::setw(1) << c;
391  }
392  o << std::endl;
393  }

395  //
396  // outputs spaces before a line based on the current indent level
397  static void indent(std::ostream& o)
398  {
399  for (uint_fast8_t i = 0; i < level; i++)
400  o << " ";
401  o.setf(std::ios::left | std::ios::showbase);
402  o << std::setfill(' ');
403  }

405  //
406  // dumps data in binary / char mode
407  //
408  void dumpData(std::ostream& o, const ui8* data, ui16 length)
409  {
410  // set stdout to not show base and pad right with 0's
411  std::ios::fmtflags f = o.flags();
412
