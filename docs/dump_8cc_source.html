350  //
351  // dumps a single data line
352  //
353  void dump_priv::dumpDataLine(std::ostream &o, const ui8 *data, ui8 length)
354  {
355  ui8 i;
356 
357  o << std::hex;
358 
359  // display the data in byte mode
360  for (i = 0; i < length; i++)
361  o << std::setw(2) << static_cast<ui16>(data[i]) << " ";
362 
363  // fill any empty spaces (if length < LINE_WIDTH)
364  for (; i < LINE_WIDTH; i++)
365  o << " ";
366 
367  // display the data in char mode
368  o << " ";
369 
370  for (i = 0; i < length; i++)
371  {
372  char c = data[i];
373 
374  // map unprintables to '.'
375  if ((c < ' ') || (c > '~'))
376  c = '.';
377 
378  o << std::setw(1) << c;
379  }
380  o << std::endl;
381  }

395  //
396  // dumps data in binary / char mode
397  //
398  void dumpData(std::ostream &o, const ui8 *data, ui16 length)
399  {
400  int num_lines = (length / LINE_WIDTH) + 1;
401 
402  // set stdout to not show base and pad right with 0's
403  std::ios::fmtflags f = o.flags();
404 
405  o.unsetf(std::ios::showbase);
406  o.setf(std::ios::right);
407  o << std::setfill('0');
408 
409  for (int i = 0, rem = length; i < num_lines; i++, rem -= LINE_WIDTH) {
410  // calculate the length of the line
411  int len = (rem < LINE_WIDTH) ? rem : LINE_WIDTH;
412 
413  o << "[" << std::dec
414  << std::setw(4)
415  << (i * LINE_WIDTH) << "] ";
416 
417  // display it
418  dumpDataLine(o, &data[i * LINE_WIDTH], len);
419  }
420  o << std::endl;
421 
422  o.flags( f );
423  }
