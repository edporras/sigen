170  bool Section::set32Bits(ui32 d)
171  {
172  assert( lengthFits(sizeof(d)) );
173 
174  *pos++ = static_cast<ui8>(d >> 24) & 0xff;
175  *pos++ = static_cast<ui8>(d >> 16) & 0xff;
176  *pos++ = static_cast<ui8>(d >> 8) & 0xff;
177  *pos++ = static_cast<ui8>(d) & 0xff;
178  data_length += sizeof(d);
179  return true;
180  }

255  //
256  // add a crc at the end of the data
257  //
258  bool Section::calcCrc()
259  {
260  ui8 *d = &data[0];
261  int i;
262  ui32 crc_accum = 0xffffffff;
263 
264  assert( lengthFits(CRC_LEN) );
265 
266  for ( int j = 0; j < data_length; j++ ) {
267  i = ( static_cast<int>(crc_accum >> 24) ^ *d++ ) & 0xff;
268  crc_accum = ( crc_accum << 8 ) ^ CrcTable[i];
269  }
270  crc = crc_accum;
271  set32Bits(crc);
272  return true;
273  }
