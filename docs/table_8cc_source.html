145  //
146  // controls the sectionable table building.. calls the virtual function
147  // writeSection() which defines how each table is written.. this function
148  // simply allocates space, calls for the write, writes the total length
149  // of the section (returned by writeSection() and crc's it
150  //
151  void PSITable::buildSections(TStream &strm) const
152  {
153  enum State_t { MALLOC_SEC, WRITE_SEC, END_TABLE };
154 
155  bool done = false;
156  ui8 cur_sec = 0;
157  ui16 sec_bytes = 0;
158  State_t state = MALLOC_SEC;
159 
160  Section *s = nullptr;
161  // this table's sections
162  std::list<Section *> table_sections;
163 
164  // add each field while it still fits in this section
165  while (!done)
166  {
167  switch (state)
168  {
169  case MALLOC_SEC:
170  s = strm.getNewSection(getMaxSectionLen());
171 
172  // add it to the temp link list - we will need to
173  // adjust some fields once we're all done with all sections
174  table_sections.push_back( s );
175 
176  state = WRITE_SEC;
177  break;
178 
179  case WRITE_SEC:
180  // write as much data as we can to this section
181  if (writeSection(*s, cur_sec, sec_bytes))
182  state = END_TABLE;
183  else {
184  // writeSection() returned 'false' which means it is not done
185  // so we increment the section count
186  cur_sec++;
187  state = MALLOC_SEC;
188  }
189  // update the section's length with the count that was added
190  s->set16Bits(1, buildLengthData(sec_bytes) + 4);
191  break;
192 
193  case END_TABLE:
194  // done with the table.. update the last_section field
195  // in all the sections
196  for (auto sp : table_sections) {
197  sp->set08Bits(7, cur_sec); // save the last_section_number
198  sp->calcCrc(); // crc the section
199  }
200  done = true;
201  break;
202  }
203  }
204  }

292  //
293  // write section data for the item
294  bool ExtPSITable::ListItem::write_section(Section& section, ui16 max_data_len,
295  ui16& sec_bytes, ui16* loop_len_ptr) const
296  {
297  ui8 header_len;
298  ui8* desc_loop_len_pos = 0;
299  ui16 d_len, desc_loop_len = 0;
300  bool exit = false, done = false;
301 
302  while (!exit)
303  {
304  switch (run.op_state)
305  {
306  case INIT:
307  // set the descriptor iterator
308  run.d_iter = descriptors.begin();
309  run.op_state = WRITE_HEAD;
310 
311  case WRITE_HEAD:
312  header_len = write_header(section) + 2;
313 
314  // save the position for the desc loop len.. we'll update it later
315  desc_loop_len_pos = section.getCurDataPosition();
316  section.set16Bits( 0 );
317 
318  // increment the byte count
319  sec_bytes += header_len;
320  if (loop_len_ptr)
321  *loop_len_ptr += header_len;
322 
323  run.op_state = (!run.d ? GET_DESC : WRITE_DESC);
324  break;
325 
326  case GET_DESC:
327  // if we have descriptors available..
328  if (run.d_iter != descriptors.end()) {
329  run.d = (*run.d_iter++).get();
330 
331  // make sure we can fit the next one
332  if ( (sec_bytes + run.d->length()) > max_data_len ) {
333  run.op_state = WRITE_HEAD;
334  exit = true;
335  break;
336  }
337  run.op_state = WRITE_DESC;
338  }
339  else {
340  // no more descriptors.. done writing this xport stream,
341  run = Context();
342  exit = done = true;
343  break;
344  }
345  break;
346 
347  case WRITE_DESC:
348  run.d->buildSections(section);
349 
350  // increment all byte counts
351  d_len = run.d->length();
352  sec_bytes += d_len;
353  if (loop_len_ptr)
354  *loop_len_ptr += d_len;
355  desc_loop_len += d_len;
356 
357  // try to get another one
358  run.op_state = GET_DESC;
359  break;
360  }
361  }
362  // write the desc loop length
363  write_desc_loop_len(section, desc_loop_len_pos, desc_loop_len);
364 
365  return done;
366  }
