66  //
67  // tests if the string can be added to the descriptor.. if not, it
68  // resizes the string, increments the descriptor length by the size
69  // and returns the resulting string
70  //
71  std::string Descriptor::incLength(const std::string& str)
72  {
73  std::string new_str(str);
74  ui16 len = new_str.length();
75 
76  // determine if the current string can fit
77  if ( !lengthFits(len) ) {
78  // nope.. truncate it to the max size
79  ui16 free_space = CAPACITY - total_length;
80 
81  new_str = new_str.substr( 0, free_space );
82  len = free_space;
83  }
84 
85  // increment the descriptor's length
86  total_length += len;
87  return new_str;
88  }

96  // adds a new language to the specified network
97  bool MultilingualTextDesc::addText(const LanguageCode& code, const std::string& text)
98  {
99  // check if we can even fit a lang code & text length byte
100  if (!incLength(Text::BASE_LEN))
101  return false;
102 
103  // try to add the length of the text.
104  // the string has been resized if it didn't fit so add the language
105  // data to the list
106  ml_text_list.emplace_back(code, incLength( text ));
107  return true;
108  }
